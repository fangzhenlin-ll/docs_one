    • 不要在 mutations 的函数中执行异步操作

    # Action 处理异步的任务

    # 方法一

    • 点击+1 时候等待一秒后执行+1 操作
    <button @click="add_count1">+1异步</button>
    1
    
actions: {
    addAsync(context){
      setTimeout(()=>{
        context.commit('add')
      },1000)
    }
  }

  // 组件中调用action
  add_count1(){
      this.$store.dispatch('addAsync')
    }

    // 如何传参
    // store.js中定义
    addNAsync(context, step) {
      setTimeout(() => {
        context.commit('add', step)
      }, 1000)
    }
  // 组件中传参
     add_count1() {
      this.$store.dispatch('addNAsync', 3)
    }

    # 方法二

    <button @click="jianyi">-1</button>

    import { mapState, mapMutations, mapActions } from 'vuex'
export default {
  data() {
    return {}
  },
  computed: {
    ...mapState(['count'])
  },
  methods: {
    ...mapMutations(['sub', 'subN']),
    ...mapActions(['subAsync', 'subNAsync']),
    jianyi() {
      this.subN(3)
    },
    jianyi1() {
      this.subAsync()
    },
    jianyi2() {
      this.subNAsync(3)
    }
  }
}
