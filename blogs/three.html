    # Express

    Express 也算一种包 基于 内置的http模块 封装出来提高开发效率的包
    添加express包 npm i express@4.17.1 版本号可以选择最新 npm i express 即可

    // 1. 导入 express
const express = require('express')
// 2. 创建 web 服务器
const app = express()

// 4. 监听客户端的 GET 和 POST 请求，并向客户端响应具体的内容
app.get('/user', (req, res) => {
  // 调用 express 提供的 res.send() 方法，向客户端响应一个 JSON 对象
  res.send({ name: 'zs', age: 20, gender: '男' })
})
app.post('/user', (req, res) => {
  // 调用 express 提供的 res.send() 方法，向客户端响应一个 文本字符串
  res.send('请求成功')
})
app.get('/', (req, res) => {
  // 通过 req.query 可以获取到客户端发送过来的 查询参数
  // 注意：默认情况下，req.query 是一个空对象
  console.log(req.query)
  res.send(req.query)
})
// 注意：这里的 :id 是一个动态的参数
app.get('/user/:ids/:username', (req, res) => {
  // req.params 是动态匹配到的 URL 参数，默认也是一个空对象
  console.log(req.params)
  res.send(req.params)
})

// 3. 启动 web 服务器
app.listen(80, () => {
  console.log('express server running at http://127.0.0.1')
})

    # express中间件

    1. 中间件的概率： 本质上就是一个function函数 只是多了个next参数 可以调用next()函数
    2. 中间件之间公用同一份req和res
    1. 要在路由之前创建中间件
    2. 客户端发过来的请求，可以连续调用多个中间件
    3. 要加 next()函数 才能执行下去
    4. 调用完 next() 就不要加其他逻辑代码了！
    5. 联系调用多个中间件时候，中间件跟路由之间是共享 req res的
    app.use((req, res, next) => {  //定义了全局生效的中间件  有请求到达服务器 必定会先经过该中间件
  console.log('这是最简单的中间件函数')
  next()
})

app.get('/', (req, res) => {
  console.log('调用了 / 这个路由')
  res.send('Home page.')
})
//当调用 127.0.0.1/的时候会得到Home page.   并输出这是最简单的中间件函数，调用了 / 这个路由

    # 共享中间件定义挂在的自定义属性

    // 这是定义全局中间件的简化形式
app.use((req, res, next) => {
  // 获取到请求到达服务器的时间
  const time = Date.now()
  // 为 req 对象，挂载自定义属性，从而把时间共享给后面的所有路由
  req.startTime = time
  next()
})

app.get('/', (req, res) => {
  res.send('Home page.' + req.startTime)
})
app.get('/user', (req, res) => {
  res.send('User page.' + req.startTime)
})

    # 局部生效的中间件

    // 1. 定义中间件函数
const mw1 = (req, res, next) => {
  console.log('调用了局部生效的中间件')
  next()
}
const mw2 = (req, res, next) => {
  console.log('调用了第二个局部生效的中间件')
  next()
}
// 2. 创建路由
app.get('/', mw1, (req, res) => {
  res.send('Home page.')
})
// 2. 创建路由
app.get('/', [mw1, mw2], (req, res) => {
  res.send('Home page.')
})

    # 中间件的分类

    1. 应用级别的中间件 通过绑定到app实力上的中间件 app.use() app.get() app.post()!
    2. 路由级别的中间件 绑定到express.Router()实例上 其实跟应用级别中间件没啥区别
    3. 错误级别的中间件 专门捕获项目中发生的异常错误 错误级别的中间件有4个参数 （err,req,res,next）!(必须在最后面定义 这样就能输出err.message 捕获错误原因)
    4. express内置的中间件
      express.static 静态托管
      express.json 解析json格式数据 app.use(express.json)
      express.urlencodeed 解析URL-encoded格式请求数据 app.use(express.urlencoded({extenden: false}))
    5. 第三方中间件 npm i body-parser
    6. 自定义中间件 当拿完全部数据后 会自动触发 req.end事件
    # 服务器发送json数据过来
    // 注意：除了错误级别的中间件，其他的中间件，必须在路由之前进行配置
// 通过 express.json() 这个中间件，解析表单中的 JSON 格式的数据
app.use(express.json())
// 通过 express.urlencoded() 这个中间件，来解析 表单中的 url-encoded 格式的数据
app.use(express.urlencoded({ extended: false }))

app.post('/user', (req, res) => {
  // 在服务器，可以使用 req.body 这个属性，来接收客户端发送过来的请求体数据
  // 默认情况下，如果不配置解析表单数据的中间件，则 req.body 默认等于 undefined
  console.log(req.body) // 输出 { name: '张三', age: 18 }
  res.send('ok')
})

app.post('/book', (req, res) => {
  // 在服务器端，可以通过 req,body 来获取 JSON 格式的表单数据和 url-encoded 格式的数据
  console.log(req.body)
  res.send('ok')
})

    # 使用querystring 模块解析请求数据 提供了parse()
    const qs = require('querystring')
app.use((req, res, next) => {
    let str = ''
    req.on('data', (chunk) => {
        str += `${chunk}`
    })

    req.on('end', () => {
        const body = qs.parse(str)
        console.log(body); //输出[Object: null prototype] { bookname: '三国\n', zuozhe: '林' }
    })
})

    # API

    const express = require('express')

const app = express();
const router = express.Router()

app.use('/api', router);
//get接口
router.get('/get', (req, res) => {
    const query = req.query

    res.send({
        status: 0, //表示处理成功  1表示处理失败
        msg: 'GET请求成功',
        data: query
    })
})

//post 接口
app.use(express.json())//解析json数据格式  加这个 req.body才有值
app.use(express.urlencoded({ extended: false }))//解析urlencoded数据格式  加这个 req.body才有值
router.post('/post', (req, res) => {
    const body = req.body
    res.send({
        status: 0, //表示处理成功  1表示处理失败
        msg: 'post请求成功',
        data: body
    })
})
